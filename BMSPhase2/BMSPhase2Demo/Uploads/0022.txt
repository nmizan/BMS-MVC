 SELECT U.AESLNO  ,U.ENTITLEFROM ,U.ENTITLEPERCENTEN ,
            U.ENTITLEPERCENTQTY ,U.ENTITLETO ,  U.ONETIMESTORAGEQTY ,U.ONETIMESTORAGEQTYEN ,U.PERCENTVAL ,
            U.TOTALSTORAGEQTY ,U.TOTALSTORAGEQTYEN ,U. MUNAME Unit,S.WEIGHT, S.MSIZE ,S.MEASUREMENT ,
            S.MATERIALNAME MatName , S.MHSCODE MHCode , S.MDESCRIPTION MDesc  ,S.MODELNM , T.MATERIALNAME ,T.MHSCODE ,T.MDESCRIPTION ,T.SPGRADE ,T.AEQTY ,T.MUNAME 
            
 FROM  ( select D.AESLNO ,D.BONDERSLNO,B.MATERIALNAME ,B.MHSCODE ,B.MDESCRIPTION ,B.SPGRADE ,A.AEQTY ,C.MUNAME 
             from  ANNUALENTLRAWMATERIAL A
             join BONDERANNUALENTITLEMENT D
             on A.BONDERSLNO = D.BONDERSLNO 
              join Materials B
             on A.MSLNO = B.MSLNO 
            join MEASUREMENTUNIT C
            on C.MUSLNO = B.MUSLNO 
            where A.AESLNO = D.AESLNO) T
            
           RIGHT JOIN
            ( select AESLNO, BONDERSLNO ,ENTITLEFROM ,ENTITLEPERCENTEN ,
            ENTITLEPERCENTQTY ,ENTITLETO ,  ONETIMESTORAGEQTY ,ONETIMESTORAGEQTYEN ,PERCENTVAL ,
            TOTALSTORAGEQTY ,TOTALSTORAGEQTYEN, J.MUNAME 
               from BONDERANNUALENTITLEMENT 
               join MEASUREMENTUNIT J
               on BONDERANNUALENTITLEMENT.TSMUSLNO = J.MUSLNO 
              
            ) U
            ON U.BONDERSLNO= T.BONDERSLNO
           FULL OUTER JOIN
           ( select P.AESLNO, P.BONDERSLNO , X.WEIGHT, X.MSIZE ,X.MEASUREMENT, Z.MATERIALNAME , Z.MHSCODE , Z.MDESCRIPTION  ,
            W.MODELNM  
            from  BONDERANNUALENTITLEMENTCOM X
            join BONDERANNUALENTITLEMENT P
            on X.BONDERSLNO = P.BONDERSLNO  
            join Materials Z
            on X.MSLNO = Z.MSLNO 
             join MACHINEINFORMATION W
             on X.MACHINESLNO  = W.MACHINESLNO
             where X.AESLNO =  P.AESLNO ) S 
 ON U.BONDERSLNO = S.BONDERSLNO
 WHERE U.AESLNO= U.AESLNO
